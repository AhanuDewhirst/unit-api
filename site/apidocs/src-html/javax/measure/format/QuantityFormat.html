030package javax.measure.format;
031
032import java.io.IOException;
033import java.text.ParsePosition;
034
035import javax.measure.Quantity;
036import javax.measure.format.ParserException;
037
038/**
039 * <p>
040 * Formats instances of {@link Quantity Quantity}.
041 * </p>
042 *
043 * @author <a href="mailto:units@catmedia.us">Werner Keil</a>
044 * @version 0.1, $Date: 2018-01-18 $
045 * @since 2.0
046 * 
047 */
048public interface QuantityFormat {
049
050  /**
051   * Formats the specified quantity into an <code>Appendable</code>.
052   *
053   * @param quantity
054   *          the quantity to format.
055   * @param dest
056   *          the appendable destination.
057   * @return the specified <code>Appendable</code>.
058   * @throws IOException
059   *           if an I/O exception occurs.
060   */
061  public Appendable format(Quantity<?> quantity, Appendable dest) throws IOException;
062
063  /**
064   * Parses a portion of the specified <code>CharSequence</code> from the specified position to produce an object. If parsing succeeds, then the index
065   * of the <code>cursor</code> argument is updated to the index after the last character used.
066   *
067   * @param csq
068   *          the <code>CharSequence</code> to parse.
069   * @param cursor
070   *          the cursor holding the current parsing index.
071   * @return the object parsed from the specified character sub-sequence.
072   * @throws IllegalArgumentException
073   *           if any problem occurs while parsing the specified character sequence (e.g. illegal syntax).
074   */
075  public Quantity<?> parse(CharSequence csq, ParsePosition cursor) throws IllegalArgumentException, ParserException;
076
077  /**
078   * Parses a portion of the specified <code>CharSequence</code> from the specified position to produce an object. If parsing succeeds, then the index
079   * of the <code>cursor</code> argument is updated to the index after the last character used.
080   *
081   * @param csq
082   *          the <code>CharSequence</code> to parse.
083   * @param cursor
084   *          the cursor holding the current parsing index.
085   * @return the object parsed from the specified character sub-sequence.
086   * @throws IllegalArgumentException
087   *           if any problem occurs while parsing the specified character sequence (e.g. illegal syntax).
088   */
089  public Quantity<?> parse(CharSequence csq) throws ParserException;
090}
